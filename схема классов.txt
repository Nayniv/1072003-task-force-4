class Task {
	проверка и запись данных новой задачи,
	проверка роли пользователя ("заказчик") и смена статуса на "завершено", смена статуса на "отменено", 
	после проверки что нет исполнителя (в статусе "новое"), смена статуса "на исполнении" и назначение автора отклика "исполнителем"
	проверка роли пользователя ("исполнитель") и смена статуса задачи на "провалено",
}

class User {
	проверка и запись данных о новом пользователе,
	получение информации о пользователе, 
	метод для добавления новой задачи, если пользователь "заказчик",
	метод для подсчета рейтинга исполнителя
	
}

class Response {
	получение информации о роли пользователя ("исполнитель") и что он ещё не откликался на это задание,
	проверка полей и запись данных в таблицу
}

class Review {
	получение информации о роли пользователя ("заказчик") и отправка данных отклика в таблицу
}
